<script>

  // ✅ FAST Google Sheets CSV export endpoint for your WebExport tab

  const CSV_URL =

    "https://docs.google.com/spreadsheets/d/1AFZrC9idkuIs2dc4uAkRK8kZiTtLtvMgR7x21Wic22o/export?format=csv&gid=1580807625";


  // Refresh + scroll tuning

  const REFRESH_MS = 4000;       // pull new standings every 4 seconds

  const SCROLL_PX_PER_TICK = 1;  // speed

  const SCROLL_TICK_MS = 35;     // smoothness

  const PAUSE_AT_TOP_MS = 1200;

  const PAUSE_AT_BOTTOM_MS = 1600;


  const tbody = document.getElementById("tbody");

  const scroller = document.getElementById("scroller");

  const refreshPill = document.getElementById("refreshPill");

  const updatedPill = document.getElementById("updatedPill");

  const errorText = document.getElementById("errorText");


  let scrollTimer = null;

  let isPaused = false;


  function parseCSV(text){

    // Small CSV parser that handles quoted commas

    const rows = [];

    let row = [], field = "", inQuotes = false;


    for (let i=0; i<text.length; i++){

      const c = text[i];

      const next = text[i+1];


      if (c === '"' && inQuotes && next === '"'){ field += '"'; i++; continue; }

      if (c === '"'){ inQuotes = !inQuotes; continue; }


      if (c === ',' && !inQuotes){ row.push(field); field=""; continue; }

      if ((c === '\n' || c === '\r') && !inQuotes){

        if (field.length || row.length){ row.push(field); rows.push(row); }

        field=""; row=[];

        if (c === '\r' && next === '\n') i++;

        continue;

      }

      field += c;

    }

    if (field.length || row.length){ row.push(field); rows.push(row); }

    return rows;

  }


  function normalize(rows){

    // Expect header: Rank, Team, Total (tolerant)

    if (!rows.length) return [];

    const header = rows[0].map(x => (x||"").trim().toLowerCase());

    const idxRank = header.indexOf("rank");

    const idxTeam = header.indexOf("team");

    const idxTotal = header.indexOf("total");


    const rI = idxRank >= 0 ? idxRank : 0;

    const tI = idxTeam >= 0 ? idxTeam : 1;

    const sI = idxTotal >= 0 ? idxTotal : 2;


    const data = [];

    for (let i=1; i<rows.length; i++){

      const r = rows[i];

      if (!r || r.length < 2) continue;

      const rank = (r[rI] ?? "").trim();

      const team = (r[tI] ?? "").trim();

      const total = (r[sI] ?? "").trim();

      if (!team) continue;

      data.push({ rank, team, total });

    }


    data.sort((a,b) => {

      const ar = parseFloat(a.rank), br = parseFloat(b.rank);

      if (Number.isFinite(ar) && Number.isFinite(br)) return ar - br;

      return (a.rank||"").localeCompare(b.rank||"");

    });


    return data;

  }


  function render(data){

    if (!data.length){

      tbody.innerHTML = `<tr><td colspan="3" style="padding:18px;">No data found. Check your WebExport tab.</td></tr>`;

      return;

    }


    tbody.innerHTML = data.map((d, i) => {

      const cls = (i===0) ? "top1" : (i===1) ? "top2" : (i===2) ? "top3" : "";

      const safeTeam = d.team

        .replaceAll("&","&amp;")

        .replaceAll("<","&lt;")

        .replaceAll(">","&gt;");

      return `

        <tr class="${cls}">

          <td class="rank">${d.rank || (i+1)}</td>

          <td class="team">${safeTeam}</td>

          <td class="total">${d.total}</td>

        </tr>`;

    }).join("");


    scroller.scrollTop = 0;

  }


  async function refresh(){

    try{

      refreshPill.textContent = "Refreshing…";

      errorText.textContent = "";


      // Bust caches

      const url = CSV_URL + (CSV_URL.includes("?") ? "&" : "?") + "t=" + Date.now();

      const res = await fetch(url, { cache: "no-store" });

      if (!res.ok) throw new Error(`Fetch failed (${res.status})`);


      const text = await res.text();

      const rows = parseCSV(text);

      const data = normalize(rows);

      render(data);


      updatedPill.textContent = "Last update: " + new Date().toLocaleTimeString();

      refreshPill.textContent = "Live";

    }catch(err){

      refreshPill.textContent = "Offline?";

      errorText.textContent = "Couldn’t load standings. Check sharing settings / Wi-Fi.";

      console.error(err);

    }

  }


  function startAutoScroll(){

    if (scrollTimer) clearInterval(scrollTimer);


    scrollTimer = setInterval(() => {

      if (isPaused) return;


      const maxScroll = scroller.scrollHeight - scroller.clientHeight;

      if (maxScroll <= 0) return;


      scroller.scrollTop += SCROLL_PX_PER_TICK;


      if (scroller.scrollTop >= maxScroll){

        isPaused = true;

        setTimeout(() => {

          scroller.scrollTop = 0;

          setTimeout(() => { isPaused = false; }, PAUSE_AT_TOP_MS);

        }, PAUSE_AT_BOTTOM_MS);

      }

    }, SCROLL_TICK_MS);

  }


  // Start

  refresh();

  setInterval(refresh, REFRESH_MS);

  startAutoScroll();
